What gets triggered automatically
1
Airbnb booking appears in Google Calendar (via iCal sync you set up once)
2
Script detects the new booking every hour, extracts the check-out date and property address
3

4

5
Host receives a heads-up email that their clean is scheduled and who is coming
💡
Cost: /bin/sh. Google Apps Script is completely free. It runs on Google's servers, checks your calendar hourly, and sends emails through your Gmail — all included in your free Google account.

Connect your Airbnb calendar
One-time setup. Takes about 3 minutes per property.
1
Get your Airbnb iCal link
1
Go to Airbnb.com → Log in → Click your profile photo → Switch to hosting
2
Go to Calendar → Click Availability settings
3
Scroll to "Connect to another calendar" → Click "Export calendar"
4
Copy the iCal link — it starts with https://www.airbnb.com/calendar/ical/
2
Add it to Google Calendar
1
Open Google Calendar → Left sidebar → Click "+" next to "Other calendars"
2
Choose "From URL" → paste your Airbnb iCal link → Click "Add calendar"
3
Your Airbnb bookings now appear in Google Calendar. Repeat for each property.
4
Note the calendar name Google gives it — you'll need it in the next step (usually "Airbnb (Property name)")
✓
Google syncs Airbnb every few hours automatically. New bookings appear in your calendar within 6 hours of being confirmed on Airbnb.

Google Apps Script — agape-dispatch.gs
// ============================================================ // AGAPE PROPERTY MANAGEMENT — Auto-Dispatch Script // Paste this into script.google.com → New Project // ============================================================ // ── YOUR CONFIGURATION ────────────────────────────────────── var CONFIG = { yourName: "James", yourEmail: "your@email.com", jobUrl: "https://agapepm.netlify.app/worker/", cleaner: { name: "Maria G.", email: "cleaner@email.com", }, properties: [ { keyword: "Maple Drive", hostEmail: "host1@email.com" }, { keyword: "Pecan Street", hostEmail: "host2@email.com" }, { keyword: "Cedar Ave", hostEmail: "host3@email.com" }, ], // Name of your Airbnb calendar in Google Calendar calendarName: "Airbnb", // How many hours before check-out to send the job alert hoursBeforeCheckout: 18, }; // ── END CONFIGURATION ──────────────────────────────────────── function checkForNewBookings() { var cal = getCalendar(CONFIG.calendarName); if (!cal) { Logger.log("Calendar not found: " + CONFIG.calendarName); return; } var now = new Date(); var lookAhead = new Date(now.getTime() + 7 * 24 * 60 * 60 * 1000); // 7 days ahead var events = cal.getEvents(now, lookAhead); var props = PropertiesService.getScriptProperties(); events.forEach(function(event) { var eventId = event.getId(); if (props.getProperty("dispatched_" + eventId)) return; // already sent var title = event.getTitle(); var start = event.getStartTime(); var end = event.getEndTime(); var desc = event.getDescription() || ""; // Only process bookings (Airbnb uses "Reserved" or guest name) if (!isBooking(title)) return; var prop = detectProperty(title + " " + desc); var checkout = formatDate(end); var checkin = formatDate(start); // Send dispatch emails sendCleanerAlert(prop, checkin, checkout, end); sendYourConfirmation(prop, checkin, checkout); if (prop.hostEmail) sendHostNotification(prop, checkin, checkout); // Mark as dispatched so we don't send again props.setProperty("dispatched_" + eventId, "true"); Logger.log("Dispatched: " + title + " → " + CONFIG.cleaner.name); }); } function isBooking(title) { var t = title.toLowerCase(); return t.indexOf("reserved") > -1 || t.indexOf("reservation") > -1 || t.indexOf("booking") > -1 || t.indexOf("airbnb") > -1 || (t.length > 2 && t.indexOf("blocked") === -1 && t.indexOf("unavailable") === -1); } function detectProperty(text) { var t = text.toLowerCase(); for (var i = 0; i < CONFIG.properties.length; i++) { if (t.indexOf(CONFIG.properties[i].keyword.toLowerCase()) > -1) { return CONFIG.properties[i]; } } return { keyword: "your property", hostEmail: "" }; } function sendCleanerAlert(prop, checkin, checkout, checkoutDate) { var subject = "🧹 New job — " + prop.keyword + " · Clean on " + checkout; var body = [ "Hi " + CONFIG.cleaner.name + ",", "", "You have a new cleaning job from Agape Property Management.", "", "📍 Property: " + prop.keyword, "📅 Guest check-in: " + checkin, "🚪 Guest check-out: " + checkout, "⏰ Please arrive: By 10:00 AM on check-out day", "", "Your job checklist:", CONFIG.jobUrl, "", "Complete the checklist on-site and submit your report when done.", "Payment will be released once the host approves your report.", "", "— " + CONFIG.yourName, " Agape Property Management", ].join("\n"); GmailApp.sendEmail(CONFIG.cleaner.email, subject, body); } function sendYourConfirmation(prop, checkin, checkout) { var subject = "✓ Dispatched — " + prop.keyword + " · " + checkout; var body = [ "Dispatch confirmation — Agape Auto-Dispatch", "", "Property: " + prop.keyword, "Check-in: " + checkin, "Check-out: " + checkout, "Cleaner: " + CONFIG.cleaner.name + " (" + CONFIG.cleaner.email + ")", "", "Job link sent to cleaner automatically.", "You will receive the report when the clean is complete.", ].join("\n"); GmailApp.sendEmail(CONFIG.yourEmail, subject, body); } function sendHostNotification(prop, checkin, checkout) { var subject = "Your clean is scheduled — " + prop.keyword; var body = [ "Hi,", "", "Your Agape cleaning service has been scheduled automatically.", "", "📍 Property: " + prop.keyword, "🚪 Check-out: " + checkout, "🧹 Cleaner: " + CONFIG.cleaner.name, "", "You will receive a full inspection report with photos once", "the clean is complete. Payment approval will follow shortly after.", "", "— " + CONFIG.yourName, " Agape Property Management", ].join("\n"); GmailApp.sendEmail(prop.hostEmail, subject, body); } function getCalendar(name) { var cals = CalendarApp.getAllCalendars(); for (var i = 0; i < cals.length; i++) { if (cals[i].getName().toLowerCase().indexOf(name.toLowerCase()) > -1) { return cals[i]; } } return null; } function formatDate(d) { return Utilities.formatDate(d, Session.getScriptTimeZone(), "EEE, MMM d yyyy"); } // ── SETUP: Run this ONCE to create the hourly trigger ──────── function setupTrigger() { // Delete any existing triggers first ScriptApp.getProjectTriggers().forEach(function(t) { ScriptApp.deleteTrigger(t); }); // Create new hourly trigger ScriptApp.newTrigger("checkForNewBookings") .timeBased() .everyHours(1) .create(); Logger.log("✓ Trigger created — script will run every hour"); }

Activate it — 3 final steps
1
In the script editor, select setupTrigger from the function dropdown → Click ▶ Run. This creates the hourly schedule.
2
Google will ask for permissions — click Review permissions → Allow. This lets the script read your calendar and send emails through your Gmail.
3
Then run checkForNewBookings once manually to test it. Check your Execution log at the bottom — you should see bookings detected and dispatched.
✓
That's it. The script now runs every hour in the background. Every new Airbnb booking automatically dispatches your cleaner — even while you sleep.
